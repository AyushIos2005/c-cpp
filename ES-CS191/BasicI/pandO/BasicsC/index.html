4. Simple and Compound Interest (maturity amount) 
#include <stdio.h>
#include <math.h>

int main() {
    double p, r;
    int t;
    printf("Enter principal (p): "); scanf("%lf", &p);
    printf("Enter annual rate (r) in percent: "); scanf("%lf", &r);
    printf("Enter time (t) in years: "); scanf("%d", &t);

    // Simple interest maturity (total amount)
    double simple_maturity = p + (p * r * t) / 100.0;

    // Compound interest (compounded annually)
    double compound_maturity = p * pow(1 + r / 100.0, t);

    printf("Simple interest maturity amount = %.2lf\n", simple_maturity);
    printf("Compound interest maturity amount = %.2lf\n", compound_maturity);
    return 0;
}

7. Real roots of a quadratic equation 
#include <stdio.h>
#include <math.h>

int main() {
    double a, b, c;
    printf("Enter coefficients a b c: ");
    scanf("%lf %lf %lf", &a, &b, &c);
    double disc = b*b - 4*a*c;
    if (a == 0) {
        printf("Not a quadratic equation (a=0)\n");
    } else if (disc > 0) {
        double r1 = (-b + sqrt(disc)) / (2*a);
        double r2 = (-b - sqrt(disc)) / (2*a);
        printf("Two distinct real roots: %.6lf and %.6lf\n", r1, r2);
    } else if (disc == 0) {
        double r = -b / (2*a);
        printf("One real root (double): %.6lf\n", r);
    } else {
        double real = -b / (2*a);
        double imag = sqrt(-disc) / (2*a);
        printf("Complex roots: %.6lf + %.6lfi and %.6lf - %.6lfi\n", real, imag, real, imag);
    }
    return 0;
}
8. Basic Calculator (+ - * /) 
#include <stdio.h>

int main() {
    int x, y;
    char op;
    printf("Enter: a operator b (e.g. 5 + 3): ");
    if (scanf("%d %c %d", &x, &op, &y) != 3) {
        printf("Invalid input\n");
        return 1;
    }
    switch (op) {
        case '+': printf("%d %c %d = %d\n", x, op, y, x + y); break;
        case '-': printf("%d %c %d = %d\n", x, op, y, x - y); break;
        case '*': printf("%d %c %d = %d\n", x, op, y, x * y); break;
        case '/':
            if (y == 0) printf("Division by zero error\n");
            else printf("%d %c %d = %.6f\n", x, op, y, (double)x / y);
            break;
        default: printf("Unsupported operator\n");
    }
    return 0;
}

10. Armstrong number check 
#include <stdio.h>
#include <math.h>

int main() {
    int num; printf("Enter an integer: "); scanf("%d", &num);
    int original = num, n = 0, temp = num;
    while (temp != 0) { temp /= 10; n++; }
    temp = num; int sum = 0;
    while (temp != 0) {
        int r = temp % 10;
        int p = 1;
        for (int i = 0; i < n; i++) p *= r;
        sum += p;
        temp /= 10;
    }
    if (sum == original) printf("%d is an Armstrong number\n", original);
    else printf("%d is not an Armstrong number\n", original);
    return 0;
}

13. Sort array in descending order (bubble sort) 
#include <stdio.h>

int main() {
    int n; printf("Enter size of array: "); scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
    for (int i = 0; i < n-1; i++) {
        for (int j = 0; j < n-1-i; j++) {
            if (arr[j] < arr[j+1]) { // swap for descending
                int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
    printf("Sorted (descending): ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");
    return 0;
}

15. Multiply 3x2 matrix by its transpose (result 3x3) 
#include <stdio.h>

int main() {
    int a[3][2];
    printf("Enter 3x2 matrix elements (row-wise):\n");
    for (int i = 0; i < 3; i++) for (int j = 0; j < 2; j++) scanf("%d", &a[i][j]);
    int b[2][3];
    // transpose 3x2 -> 2x3
    for (int i = 0; i < 3; i++) for (int j = 0; j < 2; j++) b[j][i] = a[i][j];
    int res[3][3];
    for (int i = 0; i < 3; i++) for (int j = 0; j < 3; j++) {
        res[i][j] = 0;
        for (int k = 0; k < 2; k++) res[i][j] += a[i][k] * b[k][j];
    }
    printf("Result (3x3):\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) printf("%d ", res[i][j]);
        printf("\n");
    }
    return 0;
}

17. String palindrome (with and without string.h) 
#include <stdio.h>
#include <string.h>
#include <ctype.h>

// Using string library
int is_pal_lib(const char *s) {
    int i = 0, j = strlen(s) - 1;
    while (i < j) {
        while (i < j && !isalnum((unsigned char)s[i])) i++;
        while (i < j && !isalnum((unsigned char)s[j])) j--;
        if (tolower((unsigned char)s[i]) != tolower((unsigned char)s[j])) return 0;
        i++; j--;
    }
    return 1;
}

// Without using string.h (simple version)
int is_pal_nolib(char *s) {
    int len = 0; while (s[len]) len++;
    int i = 0, j = len - 1;
    while (i < j) {
        if (s[i] != s[j]) return 0;
        i++; j--;
    }
    return 1;
}

int main() {
    char s[500];
    printf("Enter a string: \n");
    fgets(s, sizeof(s), stdin);
    if (s[strlen(s)-1] == '\n') s[strlen(s)-1] = '\0';

    printf("Using library-aware check: %s\n", is_pal_lib(s) ? "Palindrome" : "Not palindrome");

    // prepare a cleaned string (only letters) for nolib version
    char clean[500]; int idx = 0;
    for (int i = 0; s[i]; i++) if (isalnum((unsigned char)s[i])) clean[idx++] = tolower((unsigned char)s[i]);
    clean[idx] = '\0';
    printf("Without string.h check: %s\n", is_pal_nolib(clean) ? "Palindrome" : "Not palindrome");
    return 0;
}

    fgets(name, sizeof(name), stdin);
    if (name[strlen(name)-1] == '\n') name[strlen(name)-1] = '\0';

    // Split words and print initials for all but last, then full last name
    char *token = strtok(name, " ");
    char words[50][50]; int wc = 0;
    while (token != NULL) { strcpy(words[wc++], token); token = strtok(NULL, " "); }
    if (wc == 0) return 0;
    for (int i = 0; i < wc-1; i++) {
        printf("%c. ", toupper((unsigned char)words[i][0]));
    }
    printf("%s\n", words[wc-1]);
    return 0;
}
19. swap1() call by value and swap2() call by address 
#include <stdio.h>

void swap1(int a, int b) { // call by value - won't affect caller
    int t = a; a = b; b = t;
    printf("Inside swap1: a=%d b=%d\n", a, b);
}

void swap2(int *a, int *b) { // call by address - will swap
    int t = *a; *a = *b; *b = t;
}

int main() {
    int x = 5, y = 10;
    printf("Before: x=%d y=%d\n", x, y);
    swap1(x, y);
    printf("After swap1 (call by value): x=%d y=%d\n", x, y);
    swap2(&x, &y);
    printf("After swap2 (call by address): x=%d y=%d\n", x, y);
    return 0;
}

23. Student DB using struct and dynamic memory; display students with highest marks 
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Student { char name[100]; int roll; int marks; };

int main() {
    int n; printf("Enter number of students: "); scanf("%d", &n);
    struct Student *arr = malloc(n * sizeof(struct Student));
    for (int i = 0; i < n; i++) {
        printf("Enter name for student %d: ", i+1);
        getchar(); // consume newline
        fgets(arr[i].name, sizeof(arr[i].name), stdin);
        if (arr[i].name[strlen(arr[i].name)-1] == '\n') arr[i].name[strlen(arr[i].name)-1] = '\0';
        printf("Enter roll and marks: "); scanf("%d %d", &arr[i].roll, &arr[i].marks);
    }
    int maxMarks = arr[0].marks;
    for (int i = 1; i < n; i++) if (arr[i].marks > maxMarks) maxMarks = arr[i].marks;
    printf("\nStudents with highest marks (%d):\n", maxMarks);
    for (int i = 0; i < n; i++) if (arr[i].marks == maxMarks) printf("%s (Roll %d)\n", arr[i].name, arr[i].roll);
    free(arr);
    return 0;
}
24. File handling: write Profile.txt then copy to myProfile.txt 
#include <stdio.h>
#include <stdlib.h>

int main() {
    char name[100], address[200];
    printf("Enter your name: "); fgets(name, sizeof(name), stdin);
    printf("Enter your address (single line): "); fgets(address, sizeof(address), stdin);

    FILE *fp = fopen("Profile.txt", "w");
    if (!fp) { perror("Error opening Profile.txt"); return 1; }
    fprintf(fp, "Name: %s", name);
    fprintf(fp, "Address: %s", address);
    fclose(fp);

    FILE *in = fopen("Profile.txt", "r");
    FILE *out = fopen("myProfile.txt", "w");
    if (!in || !out) { perror("Error opening files"); return 1; }
    int ch;
    while ((ch = fgetc(in)) != EOF) fputc(ch, out);
    fclose(in); fclose(out);
    printf("Contents copied to myProfile.txt\n");
    return 0;
}
